country_event = {
	id = GREcrete.1
	title =  GREcrete.1.t
	desc =  GREcrete.1.d
	#picture = 

	fire_only_once = yes
	#is_triggered_only = yes

	trigger = {
		tag = OTO
		date > 1866.08.21
		182 = { is_owned_and_controlled_by = OTO } #Crete
	}

	immediate = {
	hidden_effect = {
		182 = {
			add_core_of = CET
			CET = { 
				transfer_state = PREV 
			}
		}
		1241 = {
			add_core_of = CET
			CET = { 
				transfer_state = PREV 
			}
		}
        CET = { 
			add_ideas = CET_idea_cretan_fighters
			set_cosmetic_tag = CET_Revolt_1 
			set_country_flag = CET_cretan_revolt
		}
	}
	}

	option = {
		name = GREcrete.1.a #Declare War as Ottomans 
		
		hidden_effect = {
			CET ={
				load_oob = CET_rebellion_1
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 500 
					producer = CET
				}
			}
			GRE = { news_event = {id = GREcrete.6 } } 
		}
		declare_war_on = {
			target = CET
			type = annex_everything
		}
	}
	option = {
		name = GREcrete.1.b #Take Control of the Rebels
		
		trigger = {
			is_ai = no
		}
		
		hidden_effect = {
			CET ={
				load_oob = CET_rebellion_1
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 500 
					producer = CET
				}
				change_tag_from = ROOT
			}
			GRE = { news_event = {id = GREcrete.6 } } 
		}
		declare_war_on = {
			target = CET
			type = annex_everything
		}
	}
}
