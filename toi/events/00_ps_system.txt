### YEARLY CHANGE ###
#event that calculates how much PS should be added or subtracted
country_event = {
	id = pssystem.16
	title = pssystem.16.t
	desc = pssystem.16.d
	
	mean_time_to_happen = {
		days = 365
	}
	
	trigger = {
		
	}
	
	hidden = yes
	
	immediate = {
		set_variable = {
			var = ROOT.pol_stability_yearly_factor
			value = 0
		}
		
		#has border war?
		if = {
			limit = {
				has_border_war = yes
			}
			subtract_from_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Border war: -1"
		}
		else = {
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Border war: +1"
		}
		
		#has general war?
		if = {
			limit = {
				has_war = yes
			}
			subtract_from_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: General war: -1"
		}
		else = {
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: General war: +1"
		}
		
		#has civil war?
		if = {
			limit = {
				has_civil_war = yes
			}
			subtract_from_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Civil war: -1"
		}
		else = {
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Civil war: +1"
		}
		
		#is target of coup?
		if = {
			limit = {
				is_target_of_coup = yes
			}
			subtract_from_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Target of coup: -1"
		}
		else = {
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Target of coup: +1"
		}
		
		#popular stability?
		if = {
			limit = {
				has_stability > 0.8
			}
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 2
			}
			log = "PS: Popular stability: +2"
		}
		else_if = {
			limit = {
				has_stability > 0.6
			}
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Popular stability: +1"
		}
		else_if = {
			limit = {
				has_stability > 0.4
			}
			#stab between 40 and 60 does not affect the factor
			log = "PS: Popular stability: +-0"
		}
		else_if = {
			limit = {
				has_stability > 0.2
			}
			subtract_from_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS:  Popular stability: -1"
		}
		else = {
			subtract_from_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 2
			}
			log = "PS: Popular stability: -2"
		}
		
		#party popularity?
		if = {
			limit = {
				check_variable = { ROOT.party_popularity@ruling_party > 0.8 }
			}
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 2
			}
			log = "PS: Party popularity: +2"
		}
		else_if = {
			limit = {
				check_variable = { ROOT.party_popularity@ruling_party > 0.6 }
			}
			add_to_variable = {
				var = ROOT.pol_stability_yearly_factor
				value = 1
			}
			log = "PS: Party popularity: +1"
		}
		else_if = {
			limit = {
				check_variable = { ROOT.party_popularity@ruling_party > 0.4 }
			}
			#40-60% party pupularity does not affect the factor
			
			log = "PS: Party popularity: +-0"
		}
		else_if = {
			limit = {
				check_variable = { ROOT.party_popularity@ruling_party > 0.3 }
			}
